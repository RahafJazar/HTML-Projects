  override ngOnInit(): void {

    // Load scanner policy settings

      this.THRESHOLD = data.THRESHOLD;
      this.CONTRAST = data.CONTRAST;
    });

    // Resolve dialog type and load dialog configuration
    let type = this.Type
      ? this.Type.split('.')[1]
      : this.DialogTitle.split('.')[1];

    this.initdialog = getDialogConfig(type, this);

    // Initialize UI elements
    this.InitFormGroup();
    this.InitFormList();
    this.InitDisplayColumn();
    this.ReturnAttachmentType();

    // Handle readonly / disabled dialog

    if (this.DisableDialog) {
      if (this.ActionsList[4].show) {
        this.ActionsList.forEach(element => {
          if (element.actionName != 'taskAttachments' && element.actionName != 'ViewComments')
            element.show = false
        })
      }
      else if (this.ActionsList[6].show) {
        this.ActionsList.forEach(element => {
          if (element.actionName != 'View')
            element.show = false
        })
      }
      else
        this.DisplayColumn = this.DisplayColumn.filter(col => col.field !== 'actions');
      this.FormGroupObj.disable();
    } else {
      this.buttons = this.initdialog?.buttons();
    }

    // Initialize attachments & localization
    this.InitUploadAttachments();
    this.OnChangeLang();
    if (this.Type.includes(DialogType.Disclaimer))
      this.FinishIssuingAndCopyingByReviewingWthNotAsync()
  }

        this.ReturnPartyAsync()

      this.ReturnStakeholdersAsync()

  ReturnStorePlaceAsync() {
    if (this.SelectedSystemID) {
      this.Subscription = this.GenericServiceObj.GenericService({ SelectedSystemID: this.SelectedSystemID, SelectedSystemType: this.SelectedSystemType }, this.EndPoint.ReturnStorePlaceAsync).subscribe(
        data => {
          if (data[0]) {
            this.ObjectName = data[0] as StorePlace
            this.ObjectName.FileNo = Number(data[0].FileNo)
            this.API = this.EndPoint.UpdateStorePlaceAsync
          }
          else
            this.API = this.EndPoint.NewStorePlaceAsync
          this.visible = true
        })
    }
    else {
      if (this.menuItemService.StorePlaceSaves[0])
        this.ObjectName = this.menuItemService.StorePlaceSaves[0]
      this.visible = true
    }

  CompressGroup(DataSource) {
    // Assuming `tasks` is your input list
    const tasks = DataSource

    const compressedTasks = Object.values(
      tasks.reduce((acc: Tasks | StakeholdersSave, task) => {
        if (this.Type.includes('Stakeholders')) {
          task.InnerPartyID = task.InnerPartyID ? task.InnerPartyID : task.ToOuterParty
          task.InnerPartyClassification = task.InnerPartyClassification ? task.InnerPartyClassification : (this.RouterObj.url.includes("GeneralizationCrspndns") || Number(localStorage.getItem('Type')) == CorrspondanceTypeOf.Generalization ? InnerParty.SameUnit : InnerParty.OuterParty)
        }
        const key = `${task.InnerPartyID}-${task.InnerPartyClassification}`;
        if (!acc[key]) {
          // Initialize the first occurrence
          acc[key] = { ...task };
          acc[key].GroupMember = []
          if (acc[key].InnerPartyClassification == InnerParty.GroupMember) {
            if (this.Type.includes('Tasks')) {
              acc[key].TaskTo = task.InnerPartyID
              acc[key].TaskToText = task.InnerPartyName
              acc[key].GroupMember.push({ AccountName: task.TaskToText, EntityID: task.TaskTo })
            }
            else if (this.Type.includes('Stakeholders')) {
              acc[key].TaskTo = task.InnerPartyID ? task.InnerPartyID : task.OuterPartyID
              acc[key].ArabicName = task.InnerPartyName

              acc[key].GroupMember.push({ AccountName: task.AccountName, EntityID: task.ToOuterParty ? task.ToOuterParty : task.ToInnerParty, RowId: task.IssuingAndCopyingID })
            }
          }
          else {
            acc[key].ArabicName = acc[key].AccountName ? acc[key].AccountName : acc[key].ArabicName
          }

        } else {
          if (this.Type.includes('Tasks')) {
            acc[key].GroupMember.push({ AccountName: task.TaskToText, EntityID: task.TaskTo })
          }
          else if (this.Type.includes('Stakeholders')) {
            acc[key].GroupMember.push({ AccountName: task.AccountName, EntityID: task.ToOuterParty ? task.ToOuterParty : task.ToInnerParty, RowId: task.IssuingAndCopyingID })
          }
        }

        return acc;
      }, {} as Record<string, any>)
    );

    this.DataSource = compressedTasks;

  }

  formatSize(bytes) {
    const k = 1024;
    const dm = 3;
    const sizes = this.config.translation.fileSizeTypes;
    if (bytes === 0) {
      return `0 ${sizes[0]}`;
    }

    const i = Math.floor(Math.log(bytes) / Math.log(k));
    const formattedSize = parseFloat((bytes / Math.pow(k, i)).toFixed(dm));

    return `${formattedSize} ${sizes[i]}`;
  }

  uploadFileInChunks = async (file: File, index: number) => {

    att.Description = this.subjects[index];

    }

      att.SelectedSystemType = this.SelectedSystemType;
    }
    att.Extension = att.FileExt = file.name.substr(file.name.lastIndexOf('.') + 1);
    att.ArabicName = att.FileName = file.name.substr(0, file.name.lastIndexOf('.'));
    att.Byts = await this.readFileAsDataURL(file).then(res => res); // Single base64
    att.AttachmentType = this.selectedType[index];
    att.AttachmentTypeText = this.attachmentType.find(e => e.EntityID == att.AttachmentType)?.ArabicName;
    return att;

  readFileAsDataURL = (file: Blob) => {
    return new Promise<string>((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => {
        const base64String = (reader.result as string).replace(/^data:[^;]+;base64,/, '');
        resolve(base64String);
      };
      reader.onerror = (error) => reject(error);
    });
  };

  async Action(event: any) {
    if (event.actionName == "acquise") {
      let object = event?.event?.Tasks ? JSON.parse(event?.event?.Tasks) : null
      this.AcquisTasksByDelegationAsync(object[0]?.TaskID)
    }
    else if (event.actionName == "close1") {
      this.CloseCorrespondenceAsync(event?.event?.EntityID)
    }
    else if (event.actionName == "delete") {
      let result = []
      if (this.Type.includes('InKindAttachments')) {
        result = await this.Delete(this.EndPoint.DeleteIn_kindAttachmentsAsync, event?.event, event?.event.EntityID)
        if (Array.isArray(result)) {
          this.menuItemService.In_Kinds = result
        } else {
          this.ReturnIn_kindAttachmentsAsync()
        }
      }
      else if (this.Type.includes('Beneficiaries')) {
        result = await this.Delete(this.EndPoint.DeleteEntityBeneficiaryAsync, event?.event, event?.event.EntityID)
        if (Array.isArray(result)) {
          this.menuItemService.EntityBeneficiaries = result
        } else {
          this.ReturnEntityBeneficiaryAsync()
        }
      }
      else if (this.Type.includes('Stakeholders')) {
        result = await this.DeleteStakeholders(this.EndPoint.DeleteStakeholdersAsync, event?.event, event?.event.EntityID)
        if (Array.isArray(result)) {
          this.menuItemService.StakeholdersSave = result
        } else {
          this.ReturnStakeholdersAsync()
        }
      }
      else if (this.Type.includes('Tasks')) {
        result = await this.Delete(this.EndPoint.DeleteTaskAsync, event?.event, event?.event.TaskID)
        if (Array.isArray(result)) {
          // result = this.DataSource = this.DataSource.filter(element => element != event?.event ? element : null)
          this.menuItemService.tasks = result
        }
        else if ([null, undefined, 'null', 'undefined'].includes(localStorage.getItem("TaskType"))) {
          this.ReturnTaskAsync()
        }
      }
      else if (this.Type.includes('Related')) {
        result = await this.Delete(this.EndPoint.DeleteRelated, event?.event, event?.event.RelatedEntityID)
        if (Array.isArray(result)) {
          this.menuItemService.Related = result
        } else {
          this.ReturnRelated()
        }
      }
    }

    else if (event.actionName == "ViewComments") {
      this.InitSecondDialog(event?.event?.TaskID, DialogType.ViewComments)
      // this.ShowTable = true 
    }
    else if (event.actionName == "Comment") {
      this.InitSecondDialog(event?.event?.TaskID, DialogType.Comment)

  async doScan(data, buttonID) {

    try {

      const response = await fetch("http://localhost:7080/scanActionFromWebComponent", {
        method: 'POST',
        body: data
      });

      console.log('Response status:', response.status);
      if (response.status == 200) {
        if (this.SelectedSystemID && !this.Type.includes('Tasks') && !this.Type.includes('Comment')) {
          this.ReturnAttachment()
          this.visible = true
          this.Attachments = []
          this.onHide()
        }
        else {
          // ⬇️ Get JSON here
          const json = await response.json();
          const att = new AttachmentSaves();
          if (json.content) {
            att.SelectedSystemType = this.SelectedSystemType;
            att.Extension = att.FileExt = this.ATTACHMNET_TYPE;
            const now = new Date();
            const timestamp = now.toISOString().replace(/[:]/g, "-").replace("T", "_").split(".")[0];
            att.ArabicName = att.FileName = `attachment_${timestamp}`;
            att.Byts = json.content;
            att.Description = this.description
            att.AttachmentType = this.ActionAttachmentType
            if (this.Type.includes('Tasks') || this.Type.includes('Comment')) {
              // this.ScanningAttachment.push(att)
              this.ScanningAttachment = [...this.ScanningAttachment, att];
              this.cdr.detectChanges()

            }
            else if (this.Type.includes('.AttachmentsAction')) {
              this.ScanningAttachment = [...this.ScanningAttachment, att];
              if (this.ScanningAttachment.length) {
                this.ScanningAttachment.forEach(element => {
                  this.menuItemService.AttachmentSaves?.push(element)
                })
              }
              this.menuItemService.changeData(this.MenuElements);
              this.Attachments = []
              this.onHide()
            }
            else {
              if (!this.MenuElements.attachments) this.MenuElements.attachments = []
              this.ScanningAttachment = [...this.ScanningAttachment, att];
              this.menuItemService.AttachmentSaves = this.MenuElements.attachments
              this.menuItemService.changeData(this.MenuElements);
              this.Attachments = []

            }
          }
        }

        // *** this.scann = true

        // *** this.Send = false
        // *** this.reset = true
        // *** this.disableJobsButtons = false
        // *** localStorage.setItem("hide", this.hide)
        // *** console.log("HHHHHHh", this.hide)

        // *** document.getElementById('mailDocumentDataTab:documentTable:returnedFromScanner').click();
        // *** document.getElementById('mailDocumentDataTab:documentTable:returnedFromScanner').click();
        // *** document.getElementById('mailDocumentDataTab:documentTable:returnedFromScanner').click();
        // *** document.getElementById(buttonID).click();
        const responseData = await response.status;
        return responseData;
      }
      else if (response.status == 201) {
        // *** this.disableJobsButtons = false
      }

  //format date
  formatDateToLocalISO(date: Date): string {
    const d = new Date(date);
    // Adjust for timezone offset (so date stays the same no matter the timezone)
    d.setMinutes(d.getMinutes() - d.getTimezoneOffset());
    return d.toISOString().split('T')[0];  // yyyy-MM-dd
  }

  isStatsLoading: boolean = true;
  calculateStats(data: any[]) {

    const today = new Date(); //date of today 
    this.stats = {
      total: data.length,
      urgent: data.filter(t => t.IsLate === 1 || (new Date(t.ExpiryDate) < today && t.STATUS !== 267)).length,
      read: data.filter(t => t.ReadingDateTime !== null).length,
      unread: data.filter(t => t.ReadingDateTime === null).length

    }
    this.isStatsLoading = false;
